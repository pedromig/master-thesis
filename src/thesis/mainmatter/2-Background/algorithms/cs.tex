% Constructive Search Procedure (algorithm2e pseudocode)

\KwIn{Ground Set ($\mathcal{G}$)}
\KwOut{Solution ($s$)}

\SetKwFunction{SelectComponent}{\texttt{SelectComponent}}

$s$ $\gets$ $\emptyset$\;
$\mathcal{C}$ $\gets$ $\{$  $c\in \mathcal{G} \setminus s$ $|$ $s \cup \{c\}$ \texttt{is feasible} $\}$ \;
\While{$\mathcal{C} \neq \emptyset$}{
  $c$ $\gets$ \SelectComponent{$\mathcal{C}$}\;
  $s$ $\gets$ $s \cup \{c\}$\;
  $\mathcal{C}$ $\gets$ $\{$  $c\in \mathcal{G} \setminus s$ $|$ $s \cup \{c\}$ \texttt{is feasible} $\}$ \;
}
\Return{$s$}
