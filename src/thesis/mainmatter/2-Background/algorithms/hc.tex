% Hill Climbing (algorithm2e pseudocode)

\KwIn{Solution ($s$), Objective Function ($f$)}
\KwOut{Solution ($s$)}

\SetKwFunction{Perturb}{\texttt{Perturb}}

\While{ \texttt{stopping criteria not met} }{
  $s'$ $\gets$ \Perturb{$s$}\;
  \If{$f(s') > f(s)$}{
    $s$ $\gets$ $s'$\;
  }
}
\Return{ $s$ }
